from sklearn.datasets import load_iris from sklearn.model_selection import train_test_split from sklearn.ensemble import RandomForestClassifier from sklearn.metrics import accuracy_score # 1. Load a classic dataset data = load_iris() X, y = data.data, data.target # 2. Split data into "Training" (to learn) and "Testing" (to check) X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2) # 3. Choose a Model (Random Forest is great for beginners) model = RandomForestClassifier() # 4. Train the AI model.fit(X_train, y_train) # 5. Make predictions and see how it did predictions = model.predict(X_test) print(f"AI Accuracy: {accuracy_score(y_test, predictions) * 100:.2f}%")
